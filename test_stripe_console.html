  1. Open Chrome Developer Console (Cmd+Option+J on Mac)
  2. Go to https://bankcsvconverter.com/pricing.html
  3. Run this in console to capture all messages: 
    // Intercept console messages
const originalLog = console.log;
const originalError = console.error;
const originalWarn = console.warn;

const messages = [];

console.log = function(...args) {
    const msg = args.join(' ');
    messages.push({type: 'log', message: msg, time: new Date().toISOString()});
    originalLog.apply(console, args);
    
    // Check for auth-related messages
    if (msg.includes('[Force Auth]') || msg.includes('[Stripe Complete]') || 
        msg.includes('[Auth Login Fix]') || msg.toLowerCase().includes('auth')) {
        console.warn('AUTH MESSAGE DETECTED:', msg);
    }
};

console.error = function(...args) {
    const msg = args.join(' ');
    messages.push({type: 'error', message: msg, time: new Date().toISOString()});
    originalError.apply(console, args);
};

console.warn = function(...args) {
    const msg = args.join(' ');
    messages.push({type: 'warn', message: msg, time: new Date().toISOString()});
    originalWarn.apply(console, args);
};

// Function to show all captured messages
window.showCapturedMessages = function() {
    console.warn('=== CAPTURED MESSAGES ===');
    messages.forEach(m => {
        console.warn(`[${m.time}] ${m.type.toUpperCase()}: ${m.message}`);
    });
    return messages;
};
  4. Click a Buy button and watch the console
  5. Run showCapturedMessages() to see all captured messages
